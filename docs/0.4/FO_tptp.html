Module FO_tptp

module FO_tptp: sig .. end

TPTP representation

Representation of terms and formulas for FOF (untyped TPTP)

type ty =
| Unitype 
type var = ty Var.t 
type term =
| App of ID.t * term list
| Var of var
| True
| False
| Undefined_atom of term list 
type form =
| And of form list
| Or of form list
| Not of form
| Imply of form * form
| Equiv of form * form
| Atom of term
| Eq of term * term
| Neq of term * term
| Forall of var * form
| Exists of var * form 
type role =
| R_axiom
| R_conjecture 
type statement = {
   st_name : string;
   st_role : role;
   st_form : form;
} 
type problem = {
   pb_statements : statement CCVector.ro_vector;
   pb_meta : ProblemMetadata.t;
}
